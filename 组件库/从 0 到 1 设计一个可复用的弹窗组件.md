# 从 0 到 1 设计一个可复用的弹窗组件（受控/非受控、嵌套、可访问性）
设计思路
- 核心架构：使用 Vue3 + TypeScript，基于 Composition API 实现，支持 setup 语法糖
- 模式支持：同时支持受控（通过 v-model 控制）和非受控（内部状态控制）两种模式
- 嵌套处理：通过全局层级管理器自动分配 z-index，解决嵌套弹窗层级覆盖问题
- 可访问性：遵循 WAI-ARIA 规范，支持 ESC 关闭、Tab 键焦点锁定、焦点回退
- 兼容性：使用 CSS 变量 + 降级方案，适配主流浏览器（IE11+、Chrome、Firefox、Safari）
- 生产特性：支持遮罩关闭、滚动锁定、动画过渡、按需挂载、自定义容器、销毁策略

## 创建全局层级管理器（解决嵌套和层级问题）
```ts
// src/utils/zIndexManager.ts
let baseZIndex = 1000;
const zIndexStack: number[] = [];

export const zIndexManager = {
  /**
   * 获取新的层级值（用于新弹窗）
   * @returns 新的z-index值
   */
  getNextZIndex(): number {
    const newZIndex = baseZIndex++;
    zIndexStack.push(newZIndex);
    return newZIndex;
  },

  /**
   * 释放层级值（弹窗关闭时）
   * @param index 要释放的z-index值
   */
  releaseZIndex(index: number): void {
    const idx = zIndexStack.indexOf(index);
    if (idx > -1) {
      zIndexStack.splice(idx, 1);
    }

    // 栈空时重置基础值，防止数值过大
    if (zIndexStack.length === 0) {
      baseZIndex = 1000;
    }
  },

  /**
   * 获取当前最高层级
   * @returns 当前最高z-index值
   */
  getCurrentMax(): number {
    return zIndexStack.length > 0 ? Math.max(...zIndexStack) : 1000;
  }
};
```
## 弹窗组件核心实现（Modal.vue）
```vue
<template>
  <!-- 动态挂载的弹窗容器 -->
  <teleport :to="getContainer()">
    <div
      v-if="isShow"
      ref="overlayRef"
      class="modal-overlay"
      :class="{ 'modal-overlay-hidden': !visible }"
      :style="{ zIndex: zIndex }"
      @click="handleMaskClick"
    >
      <!-- 弹窗主体 -->
      <div
        ref="modalRef"
        class="modal-content"
        :class="[
          className,
          { 'modal-content-hidden': !visible }
        ]"
        :style="style"
        role="dialog"
        aria-modal="true"
        aria-labelledby="modal-title"
        @click.stop
      >
        <!-- 弹窗头部 -->
        <div class="modal-header" v-if="title || !hideClose">
          <h2 id="modal-title" class="modal-title" v-if="title">
            <slot name="title">{{ title }}</slot>
          </h2>
          <button
            v-if="!hideClose"
            class="modal-close"
            aria-label="关闭弹窗"
            @click="handleClose"
          >
            ×
          </button>
        </div>

        <!-- 弹窗内容 -->
        <div class="modal-body" role="document">
          <slot>{{ children }}</slot>
        </div>

        <!-- 弹窗底部 -->
        <div class="modal-footer" v-if="showFooter">
          <slot name="footer">
            <button class="modal-btn modal-btn-cancel" @click="handleClose">
              取消
            </button>
            <button class="modal-btn modal-btn-ok" @click="handleOk">
              确认
            </button>
          </slot>
        </div>
      </div>
    </div>
  </teleport>
</template>

<script setup lang="ts">
import {
  ref,
  computed,
  watch,
  onMounted,
  onUnmounted,
  onBeforeUnmount,
  defineProps,
  defineEmits,
  defineExpose,
  type Ref,
  type CSSProperties
} from 'vue';
import { zIndexManager } from '@/utils/zIndexManager';

// 定义Props
const props = defineProps<{
  // 受控模式：通过v-model绑定
  modelValue?: boolean;
  // 非受控模式：默认显示状态
  defaultVisible?: boolean;
  // 弹窗标题
  title?: string;
  // 弹窗内容（备用，优先使用slot）
  children?: string;
  // 点击遮罩是否关闭
  maskClosable?: boolean;
  // 是否显示遮罩
  mask?: boolean;
  // 是否隐藏关闭按钮
  hideClose?: boolean;
  // 自定义类名
  className?: string;
  // 自定义样式
  style?: CSSProperties;
  // 挂载容器
  getContainer?: () => HTMLElement;
  // 是否锁定背景滚动
  lockScroll?: boolean;
  // 是否显示底部按钮栏
  showFooter?: boolean;
}>();

// 定义Emits
const emit = defineEmits<{
  (e: 'update:modelValue', value: boolean): void;
  (e: 'close'): void;
  (e: 'ok'): void;
}>();

// 默认值处理
const defaultProps = {
  defaultVisible: false,
  maskClosable: true,
  mask: true,
  hideClose: false,
  lockScroll: true,
  getContainer: () => document.body,
  showFooter: true,
  className: '',
  style: {} as CSSProperties
};

// 合并props和默认值
const mergedProps = { ...defaultProps, ...props };

// 区分受控/非受控模式
const isControlled = 'modelValue' in props;
const innerVisible = ref(mergedProps.defaultVisible);

// 计算最终显示状态
const visible = computed({
  get(): boolean {
    return isControlled ? (props.modelValue as boolean) : innerVisible.value;
  },
  set(value: boolean) {
    if (!isControlled) {
      innerVisible.value = value;
    }
    emit('update:modelValue', value);
  }
});

// 计算是否显示（用于动画过渡）
const isShow = ref(false);
watch(visible, (newVal) => {
  if (newVal) {
    isShow.value = true;
    // 分配z-index
    zIndexRef.value = zIndexManager.getNextZIndex();
    // 锁定滚动
    if (mergedProps.lockScroll) {
      lockBodyScroll();
    }
    // 保存最后焦点元素
    lastFocusRef.value = document.activeElement as HTMLElement;
    // 焦点锁定
    nextTick(() => {
      trapFocus();
    });
  } else {
    // 触发关闭动画
    nextTick(() => {
      // 动画结束后隐藏
      setTimeout(() => {
        isShow.value = false;
        // 释放z-index
        zIndexManager.releaseZIndex(zIndexRef.value);
        // 解锁滚动
        if (mergedProps.lockScroll) {
          unlockBodyScroll();
        }
        // 恢复焦点
        if (lastFocusRef.value) {
          lastFocusRef.value.focus();
        }
      }, 300); // 匹配CSS动画时长
    });
  }
});

// 引用定义
const modalRef = ref<HTMLDivElement>(null);
const overlayRef = ref<HTMLDivElement>(null);
const zIndexRef = ref(0);
const lastFocusRef = ref<HTMLElement | null>(null);
const zIndex = computed(() => zIndexRef.value);

// 关闭弹窗方法
const handleClose = () => {
  visible.value = false;
  emit('close');
};

// 确认按钮方法
const handleOk = () => {
  emit('ok');
};

// 点击遮罩关闭
const handleMaskClick = (e: MouseEvent) => {
  if (mergedProps.maskClosable && e.target === overlayRef.value) {
    handleClose();
  }
};

// 锁定body滚动
const lockBodyScroll = () => {
  const body = document.body;
  const scrollBarWidth = window.innerWidth - document.documentElement.clientWidth;
  body.style.overflow = 'hidden';
  body.style.paddingRight = `${scrollBarWidth}px`;
};

// 解锁body滚动
const unlockBodyScroll = () => {
  const body = document.body;
  body.style.overflow = '';
  body.style.paddingRight = '';
};

// 焦点锁定（可访问性）
const trapFocus = () => {
  if (!modalRef.value) return;
  
  const focusableElements = modalRef.value.querySelectorAll(
    'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
  );
  
  const firstElement = focusableElements[0] as HTMLElement;
  const lastElement = focusableElements[focusableElements.length - 1] as HTMLElement;
  
  if (firstElement) firstElement.focus();
  
  // Tab键循环焦点
  modalRef.value.addEventListener('keydown', (e: KeyboardEvent) => {
    if (e.key === 'Tab') {
      if (e.shiftKey && document.activeElement === firstElement) {
        e.preventDefault();
        lastElement.focus();
      } else if (!e.shiftKey && document.activeElement === lastElement) {
        e.preventDefault();
        firstElement.focus();
      }
    }
    
    // ESC键关闭
    if (e.key === 'Escape') {
      handleClose();
    }
  });
};

// 键盘事件监听
onMounted(() => {
  if (visible.value) {
    zIndexRef.value = zIndexManager.getNextZIndex();
    if (mergedProps.lockScroll) lockBodyScroll();
  }
});

// 清理资源
onBeforeUnmount(() => {
  if (mergedProps.lockScroll) unlockBodyScroll();
  zIndexManager.releaseZIndex(zIndexRef.value);
});

// 暴露组件方法
defineExpose({
  open: () => {
    if (isControlled) {
      console.warn('受控模式下请通过修改v-model值打开弹窗');
      return;
    }
    visible.value = true;
  },
  close: handleClose
});
</script>
```

## 组件使用示例（App.vue）
```vue
<template>
  <div class="app">
    <button @click="openModal1">打开弹窗1（非受控）</button>
    <button @click="openModal2">打开弹窗2（受控）</button>

    <!-- 非受控弹窗 -->
    <Modal
      ref="modal1Ref"
      title="非受控弹窗示例"
      default-visible="false"
      @close="handleModal1Close"
      @ok="handleModal1Ok"
    >
      <p>这是一个非受控弹窗，内部状态管理显隐</p>
      <button @click="openNestedModal">打开嵌套弹窗</button>
    </Modal>

    <!-- 受控弹窗 -->
    <Modal
      v-model="modal2Visible"
      title="受控弹窗示例"
      @close="handleModal2Close"
      @ok="handleModal2Ok"
    >
      <p>这是一个受控弹窗，通过v-model控制显隐</p>
    </Modal>

    <!-- 嵌套弹窗 -->
    <Modal
      v-model="nestedModalVisible"
      title="嵌套弹窗示例"
      @close="nestedModalVisible = false"
    >
      <p>这是嵌套在弹窗1中的子弹窗，层级会自动提升</p>
    </Modal>
  </div>
</template>

<script setup lang="ts">
import { ref } from 'vue';
import Modal from './components/Modal.vue';

// 非受控弹窗
const modal1Ref = ref();
const openModal1 = () => {
  modal1Ref.value.open();
};
const handleModal1Close = () => {
  console.log('非受控弹窗关闭');
};
const handleModal1Ok = () => {
  console.log('非受控弹窗确认');
  modal1Ref.value.close();
};

// 受控弹窗
const modal2Visible = ref(false);
const openModal2 = () => {
  modal2Visible.value = true;
};
const handleModal2Close = () => {
  console.log('受控弹窗关闭');
  modal2Visible.value = false;
};
const handleModal2Ok = () => {
  console.log('受控弹窗确认');
  modal2Visible.value = false;
};

// 嵌套弹窗
const nestedModalVisible = ref(false);
const openNestedModal = () => {
  nestedModalVisible.value = true;
};
</script>
```
