# C 端场景下的瀑布流布局

C 端瀑布流需满足
- 自适应布局：适配移动端 / PC 端不同屏幕宽度；
- 懒加载：图片按需加载，避免首屏加载过多资源
- 无限滚动：滚动到底部自动加载下一页；
- 性能优化：避免重排重绘、图片加载失败兜底；
- 体验优化：加载中占位、加载完成平滑过渡、无更多数据提示

## 完整实现代码
```vue
<template>
  <div class="waterfall-wrapper" ref="wrapperRef">
    <!-- 瀑布流容器 -->
    <div class="waterfall-container" ref="containerRef">
      <!-- 列容器 -->
      <div
        v-for="(column, index) in columnList"
        :key="index"
        class="waterfall-column"
        :style="{
          width: `${columnWidth}px`,
          marginRight: index < columnList.length - 1 ? `${gap}px` : '0'
        }"
      >
        <!-- 列表项 -->
        <div
          v-for="item in column"
          :key="item.id"
          class="waterfall-item"
          ref="itemRefs"
          :data-id="item.id"
        >
          <!-- 骨架屏 -->
          <div v-if="item.loading" class="waterfall-skeleton">
            <div class="skeleton-img"></div>
            <div class="skeleton-text"></div>
          </div>
          <!-- 内容区 -->
          <div v-else class="waterfall-content">
            <img
              ref="imgRefs"
              :data-src="item.imgUrl"
              class="waterfall-img"
              @load="handleImgLoad(item)"
              @error="handleImgError(item)"
              alt="瀑布流内容"
            />
            <div class="waterfall-desc">{{ item.title }}</div>
          </div>
        </div>
      </div>
    </div>

    <!-- 加载状态 -->
    <div class="waterfall-status">
      <span v-if="loading">加载中...</span>
      <span v-else-if="noMore">已加载全部内容</span>
    </div>
  </div>
</template>

<script setup>
import { ref, reactive, onMounted, onUnmounted, nextTick, watch } from 'vue'
import { debounce } from 'lodash-es' // 防抖（也可手写，见下方备注）

// Props 定义
const props = defineProps({
  list: {
    type: Array,
    default: () => [],
    required: true
  },
  gap: {
    type: Number,
    default: 10 // 列间距/项间距（px）
  },
  baseColumnWidth: {
    type: Number,
    default: 240 // 基础列宽（PC端）
  },
  loading: {
    type: Boolean,
    default: false
  },
  noMore: {
    type: Boolean,
    default: false
  }
})

const emit = defineEmits(['loadMore'])

// 核心响应式数据
const wrapperRef = ref(null) // 外层容器
const containerRef = ref(null) // 瀑布流容器
const itemRefs = ref([]) // 列表项DOM
const imgRefs = ref([]) // 图片DOM
const columnList = reactive([]) // 列数据列表
const columnWidth = ref(0) // 实际列宽
const columnHeights = ref([]) // 列高度数组
let lazyLoadObserver = null // 懒加载观察者
let resizeHandler = null // 窗口resize处理函数

// ===================== 核心方法 =====================
/**
 * 初始化瀑布流布局
 */
const initWaterfall = debounce(async () => {
  if (!wrapperRef.value || !containerRef.value) return

  // 1. 获取容器宽度（适配移动端）
  const wrapperWidth = wrapperRef.value.clientWidth
  // 2. 计算列数（最少1列）
  const columnCount = Math.max(
    1,
    Math.floor((wrapperWidth + props.gap) / (props.baseColumnWidth + props.gap))
  )
  // 3. 计算实际列宽（去除间距后均分）
  columnWidth.value = (wrapperWidth - (columnCount - 1) * props.gap) / columnCount

  // 4. 重置列数据和高度
  columnList.length = 0
  columnHeights.value = new Array(columnCount).fill(0)
  for (let i = 0; i < columnCount; i++) {
    columnList.push([])
  }

  // 5. 分配数据到列
  await nextTick()
  renderWaterfall(props.list)

  // 6. 初始化图片懒加载
  initLazyLoad()
}, 100) // 防抖100ms，避免resize频繁触发

/**
 * 分配数据到最短列（核心逻辑）
 * @param {Array} list - 数据源
 */
const renderWaterfall = (list) => {
  if (!list.length) return

  list.forEach(item => {
    // 标记加载中（显示骨架屏）
    if (item.loading === undefined) {
      item.loading = true
    }
    // 找到当前最短列
    const minHeight = Math.min(...columnHeights.value)
    const minIndex = columnHeights.value.indexOf(minHeight)
    // 添加到最短列
    columnList[minIndex].push(item)
    // 预占高度（避免布局抖动，取图片平均高度）
    columnHeights.value[minIndex] += 200 + props.gap
  })
}

/**
 * 初始化图片懒加载（兼容降级）
 */
const initLazyLoad = () => {
  // 销毁旧的观察者
  if (lazyLoadObserver) {
    lazyLoadObserver.disconnect()
  }

  // 方案1：支持IntersectionObserver（现代浏览器）
  if ('IntersectionObserver' in window) {
    lazyLoadObserver = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const img = entry.target
          const src = img.dataset.src
          if (src) {
            img.src = src
            lazyLoadObserver.unobserve(img)
          }
        }
      })
    }, {
      rootMargin: '100px 0px', // 提前100px加载
      threshold: 0.01
    })

    // 监听所有图片
    imgRefs.value.forEach(img => {
      lazyLoadObserver.observe(img)
    })
  } else {
    // 方案2：降级方案（scroll监听 + 防抖）
    const lazyLoadHandler = debounce(() => {
      imgRefs.value.forEach(img => {
        if (img.src) return
        const rect = img.getBoundingClientRect()
        if (rect.top < window.innerHeight + 100) {
          img.src = img.dataset.src
        }
      })
    }, 200)

    window.addEventListener('scroll', lazyLoadHandler)
    // 组件卸载时移除监听
    onUnmounted(() => {
      window.removeEventListener('scroll', lazyLoadHandler)
    })
    // 首次触发
    lazyLoadHandler()
  }
}

/**
 * 图片加载完成处理
 * @param {Object} item - 数据项
 */
const handleImgLoad = async (item) => {
  item.loading = false
  await nextTick()

  // 找到当前项DOM
  const itemEl = itemRefs.value.find(el => el.dataset.id === item.id)
  if (!itemEl) return

  // 找到所在列
  const columnIndex = columnList.findIndex(col => col.includes(item))
  if (columnIndex === -1) return

  // 更新列高度（使用requestAnimationFrame避免重排）
  requestAnimationFrame(() => {
    // 取列最后一项的底部位置作为列高度（最准确）
    const lastItem = columnList[columnIndex].at(-1)
    const lastItemEl = itemRefs.value.find(el => el.dataset.id === lastItem.id)
    if (lastItemEl) {
      columnHeights.value[columnIndex] = lastItemEl.offsetTop + lastItemEl.offsetHeight + props.gap
    }
  })
}

/**
 * 图片加载失败兜底
 * @param {Object} item - 数据项
 */
const handleImgError = (item) => {
  item.loading = false
  // 找到图片DOM并替换为兜底图
  const imgEl = imgRefs.value.find(img => {
    const parent = img.closest('[data-id]')
    return parent?.dataset.id === item.id
  })
  if (imgEl) {
    imgEl.src = '/images/default.png' // 替换为你的兜底图地址
  }
}

/**
 * 滚动监听（无限加载）
 */
const handleScroll = debounce(() => {
  if (props.loading || props.noMore) return

  // 计算滚动位置（兼容各浏览器）
  const scrollTop = document.documentElement.scrollTop || document.body.scrollTop
  const windowHeight = window.innerHeight
  const scrollHeight = document.documentElement.scrollHeight

  // 滚动到底部前200px触发加载
  if (scrollTop + windowHeight >= scrollHeight - 200) {
    emit('loadMore')
  }
}, 150)

// ===================== 生命周期 =====================
onMounted(() => {
  // 初始化布局
  initWaterfall()

  // 监听窗口resize
  resizeHandler = debounce(initWaterfall, 100)
  window.addEventListener('resize', resizeHandler)

  // 监听滚动
  window.addEventListener('scroll', handleScroll)

  // 初始化时触发一次懒加载
  nextTick(initLazyLoad)
})

onUnmounted(() => {
  // 清理监听
  if (resizeHandler) {
    window.removeEventListener('resize', resizeHandler)
  }
  window.removeEventListener('scroll', handleScroll)

  // 销毁懒加载观察者
  if (lazyLoadObserver) {
    lazyLoadObserver.disconnect()
  }
})

// 监听数据源变化，重新渲染
watch(
  () => props.list,
  () => {
    initWaterfall()
  },
  { deep: true }
)

// 监听loading状态，停止滚动触发
watch(
  () => props.loading,
  (val) => {
    if (val) return
    // 加载完成后重新检查懒加载
    initLazyLoad()
  }
)
</script>
```
